# 前端

## 安装 Headless UI（必须）
npm install @headlessui/vue

## 同时强烈建议一起装 Heroicons（因为你用了 Bars3Icon、XMarkIcon）
npm install @heroicons/vue


# Go
wget https://go.dev/dl/go1.25.2.linux-amd64.tar.gz
sudo tar -C /usr/local -xzf go1.25.2.linux-amd64.tar.gz

echo "export PATH=$PATH:/usr/local/go/bin" >> ~/.profile
echo "export GOPATH=$HOME/go" >> ~/.profile # 可选：设置Go工作空间
echo "export PATH=$PATH:$GOPATH/bin" >> ~/.profile # 可选：将Go工作空间的bin目录加入PATH

source ~/.profile
